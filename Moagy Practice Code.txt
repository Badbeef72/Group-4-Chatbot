import random # Importing the module I need to allow me to chose a random value, whether it be str, int or bool


detectableGreetings = ("Hello", "Greetings", "Hi", "Bonjour", "Ciao", "Hey", "What's up", "Sup") # This is a list of greetings the bot will pick up on and respond to

listResponses = ["Hello", "Greetings", "Hi", "Bonjour", "Ciao", "Hey", "What's up", "Sup"] # This is a list of possible responses the bot may choose from when responding to the greeting

def checkForGreeting(sentence): # Defining the function to check for a greeting and respond to it
    for word in sentence.words: # Loop to make the bot look for a greeting
        if word in detectableGreetings: # Conditional if statement; checking to see if the word inputted by the user is in the list of keywords assigned for the bot to pick up on
            print(random.choice(listResponses)) # Print function; printing a random greeting in response to the users greeting
